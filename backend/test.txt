   const res = await fetch(this.configService.get('PROVIDER_URL')!, {
        body: JSON.stringify({
          hall: this.configService.get('PROVIDER_HALL_ID'),
          key: this.configService.get('PROVIDER_HALL_KEY'),
          cmd: 'getGamesList',
        }),
        method: 'POST',
      });
      const data = (await res.json()).content['amatic'];
      const dataDemo1 = data.filter((game) => game.demo === '1');
      dataDemo1.map(async (game) => {
        await this.db.insert(gamesTable).values({
          name: game.name,
          provider: game.title,
          slug: game.name,
          providerGameId: game.id,
          img: game.img,
        });
      });
      return data;
